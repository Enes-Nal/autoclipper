# Cutly Mobile Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a mobile-optimized page at `/mobile` with a 3-step flow (download → pick template → export) and a share code system to transfer template designs from desktop to mobile.

**Architecture:** A new `frontend/mobile.html` is served by a single new Flask route. It reuses all existing backend APIs with no backend changes except that one route. Template share codes are base64-encoded template JSON generated client-side on desktop and decoded client-side on mobile — no server involvement.

**Tech Stack:** Vanilla HTML/CSS/JS, Flask (Python), existing `/api/download`, `/api/export`, `/api/templates` endpoints.

// ── Step 1: Download ───────────────────────────────────────────
async function startDownload() {
  const url = document.getElementById('url-input').value.trim();
  if (!url) { showError('dl-error', 'Please enter a video URL.'); return; }

  hideError('dl-error');
  showProgress('dl-progress', 'dl-progress-bar', 'dl-progress-label', 0, 'Starting download…');
  document.getElementById('dl-btn').disabled = true;

  let res;
  try {
    res = await fetch('/api/download', {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify({url})
    }).then(r => r.json());
  } catch(e) {
    showError('dl-error', 'Network error — please try again.');
    document.getElementById('dl-btn').disabled = false;
    return;
  }

  if (res.error) {
    showError('dl-error', res.error);
    document.getElementById('dl-btn').disabled = false;
    hideProgress('dl-progress');
    return;
  }

  listenSSE(
    `/api/download/${res.job_id}/progress`,
    (msg) => {
      if (msg.type === 'progress') {
        const pct = msg.percent ?? 0;
        showProgress('dl-progress', 'dl-progress-bar', 'dl-progress-label', pct, `Downloading… ${Math.round(pct)}%`);
      } else if (msg.type === 'done') {
        _videoPath = msg.video_path;
        hideProgress('dl-progress');
        setStepDone(document.getElementById('step1'));
        unlockStep(document.getElementById('step2'));
        loadTemplates();
      } else if (msg.type === 'error') {
        showError('dl-error', msg.message || 'Download failed.');
        hideProgress('dl-progress');
        document.getElementById('dl-btn').disabled = false;
      }
    }
  );
}

async function handleFileUpload(input) {
  const file = input.files[0];
  if (!file) return;

  hideError('dl-error');
  showProgress('dl-progress', 'dl-progress-bar', 'dl-progress-label', 0, 'Uploading…');

  const fd = new FormData();
  fd.append('file', file);

  let res;
  try {
    res = await fetch('/api/upload-video', {method:'POST', body:fd}).then(r=>r.json());
  } catch(e) {
    showError('dl-error', 'Upload failed — please try again.');
    hideProgress('dl-progress');
    return;
  }

  if (res.error) {
    showError('dl-error', res.error);
    hideProgress('dl-progress');
    return;
  }

  _videoPath = res.path;
  hideProgress('dl-progress');
  document.getElementById('file-zone').textContent = '✓ ' + file.name;
  setStepDone(document.getElementById('step1'));
  unlockStep(document.getElementById('step2'));
  loadTemplates();
}

// ── Step 2: Templates ──────────────────────────────────────────
async function loadTemplates() {
  const grid = document.getElementById('tpl-grid');
  grid.innerHTML = '<div class="m-tpl-loading">Loading templates…</div>';

  let templates;
  try {
    templates = await fetch('/api/templates').then(r => r.json());
  } catch(e) {
    grid.innerHTML = '<div class="m-tpl-loading">Failed to load templates.</div>';
    return;
  }

  renderTemplateGrid(templates);
}

function renderTemplateGrid(templates) {
  const grid = document.getElementById('tpl-grid');
  let html = '';

  // Imported template at top if present
  if (_importedTemplate) {
    html += `<div class="m-tcard selected" id="tcard-imported" onclick="selectTemplate(null, 'imported')">
      <div class="m-tcard-thumb">⬇</div>
      <div class="m-tcard-name">Imported</div>
    </div>`;
  }

  templates.forEach(t => {
    const emoji = t.builtin ? '🎬' : '📄';
    html += `<div class="m-tcard" id="tcard-${t.file}" onclick="selectTemplate('${t.file}', null)">
      <div class="m-tcard-thumb">${emoji}</div>
      <div class="m-tcard-name">${t.name}</div>
    </div>`;
  });

  grid.innerHTML = html || '<div class="m-tpl-loading">No templates found.</div>';

  // If imported template was already selected, keep it active
  if (_importedTemplate && _selectedTemplate === _importedTemplate) {
    document.getElementById('tcard-imported')?.classList.add('selected');
  }
}

async function selectTemplate(file, special) {
  // Deselect all
  document.querySelectorAll('.m-tcard').forEach(c => c.classList.remove('selected'));

  if (special === 'imported' && _importedTemplate) {
    _selectedTemplate = _importedTemplate;
    document.getElementById('tcard-imported')?.classList.add('selected');
  } else {
    try {
      const data = await fetch(`/api/templates/${file}`).then(r => r.json());
      _selectedTemplate = data;
      document.getElementById(`tcard-${file}`)?.classList.add('selected');
    } catch(e) {
      return;
    }
  }

  unlockStep(document.getElementById('step3'));
}

async function importCode() {
  const raw = document.getElementById('paste-input').value.trim();
  const errEl = document.getElementById('paste-error');
  errEl.classList.remove('show');

  if (!raw.startsWith('CutlyV1:')) {
    errEl.classList.add('show');
    return;
  }

  let tpl;
  try {
    tpl = JSON.parse(atob(raw.slice('CutlyV1:'.length)));
    if (!tpl || typeof tpl !== 'object') throw new Error('not an object');
  } catch(e) {
    errEl.classList.add('show');
    return;
  }

  _importedTemplate = tpl;
  document.getElementById('paste-modal').classList.remove('open');

  // Re-render template grid with imported at top, auto-selected
  const templates = await fetch('/api/templates').then(r => r.json()).catch(() => []);
  renderTemplateGrid(templates);
  selectTemplate(null, 'imported');
}

// ── Step 3: Export ─────────────────────────────────────────────
async function startExport() {
  if (!_videoPath || !_selectedTemplate) return;

  hideError('export-error');
  document.getElementById('download-link').classList.remove('show');
  document.getElementById('ios-tip').classList.remove('show');
  showProgress('export-progress', 'export-progress-bar', 'export-progress-label', 0, 'Starting export…');
  document.getElementById('export-btn').disabled = true;

  let res;
  try {
    res = await fetch('/api/export', {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify({video_path: _videoPath, template: _selectedTemplate})
    }).then(r => r.json());
  } catch(e) {
    showError('export-error', 'Network error — please try again.');
    hideProgress('export-progress');
    document.getElementById('export-btn').disabled = false;
    return;
  }

  if (res.error) {
    showError('export-error', res.error);
    hideProgress('export-progress');
    document.getElementById('export-btn').disabled = false;
    return;
  }

  let exportDone = false;
  listenSSE(
    `/api/export/${res.job_id}/progress`,
    (msg) => {
      if (msg.type === 'progress') {
        // Export progress comes as ffmpeg lines; show indeterminate bar
        const pct = msg.percent ?? 50;
        showProgress('export-progress', 'export-progress-bar', 'export-progress-label', pct, 'Exporting…');
      } else if (msg.type === 'done') {
        exportDone = true;
        hideProgress('export-progress');
        document.getElementById('export-btn').disabled = false;

        const url = msg.download_url || `/api/exports/${msg.filename}`;
        const anchor = document.getElementById('download-anchor');
        anchor.href = url;
        anchor.download = msg.filename || 'export.mp4';
        document.getElementById('download-link').classList.add('show');
        document.getElementById('ios-tip').classList.add('show');
        setStepDone(document.getElementById('step3'));
      } else if (msg.type === 'error') {
        hideProgress('export-progress');
        showError('export-error', msg.message || 'Export failed.');
        document.getElementById('export-btn').disabled = false;
      }
    },
    // Fallback: if SSE drops after 30s with no done, show "still processing"
    () => {
      if (!exportDone) {
        showProgress('export-progress', 'export-progress-bar', 'export-progress-label', 70, 'Still processing… please wait.');
      }
    }
  );
}

// ── SSE helper ─────────────────────────────────────────────────
function listenSSE(url, onMsg, onClose) {
  const es = new EventSource(url);
  es.onmessage = (e) => {
    let msg;
    try { msg = JSON.parse(e.data); } catch { return; }
    onMsg(msg);
    if (msg.type === 'done' || msg.type === 'error') { es.close(); if(onClose) onClose(); }
  };
  es.onerror = () => { es.close(); if(onClose) onClose(); };
}

async function shareTemplate(file) {
  try {
    const tpl = await fetch(`/api/templates/${file}`).then(r => r.json());
    const code = 'CutlyV1:' + btoa(JSON.stringify(tpl));
    await navigator.clipboard.writeText(code);
    // Brief visual feedback: swap button text
    const btn = document.querySelector(`.tcard[data-tpl-file="${file}"] .tcard-share`);
    if (btn) { btn.textContent = 'Copied!'; setTimeout(() => { btn.textContent = 'Share'; }, 1800); }
  } catch(e) {
    alert('Could not copy share code: ' + e.message);
  }
}
